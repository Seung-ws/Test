USE EUNDB;
CREATE TABLE MEMBERS(
	MEMBER_UID VARCHAR(36) NOT NULL,
	MEMBER_GID VARCHAR(36) DEFAULT NULL,
	MEMBER_ID VARCHAR(20),
	MEMBER_USERNAME VARCHAR(20),
	MEMBER_PASSWORD VARCHAR(64) NOT NULL,
	MEMBER_EMAIL VARCHAR(50) DEFAULT NULL,
	MEMBER_IMAGE VARCHAR(100) DEFAULT NULL,
	MEMBER_CREATEDATE TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
	CONSTRAINT UK_MEMBERS_MEMBER_UID UNIQUE(MEMBER_UID),
	CONSTRAINT PK_MEMBERS_MEMBER_ID PRIMARY KEY(MEMBER_ID)
);
CREATE UNIQUE INDEX INDEX_MEMBERS ON MEMBERS(MEMBER_ID);
